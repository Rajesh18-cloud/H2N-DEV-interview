Python script to convert your given XML data into a Python dictionary and then print it as JSON:
import xml.etree.ElementTree as ET
import json

# XML data
xml_data = """
<Orders>
    <Order>
        <OrderID>141</OrderID>
        <Customer>
            <CustomerID>141</CustomerID>
            <Name>Customer 141</Name>
        </Customer>
        <OrderDate>2024-10-15T10:21:00</OrderDate>
        <Products/>
        <TotalAmount>400.37</TotalAmount>
    </Order>
</Orders>
"""

# Function to convert XML to dictionary
def xml_to_dict(element):
    if len(element) == 0:
        return element.text
    return {child.tag: xml_to_dict(child) for child in element}

# Parse XML data
root = ET.fromstring(xml_data)
orders_dict = xml_to_dict(root)

# Convert dictionary to JSON
orders_json = json.dumps(orders_dict, indent=4)

# Print JSON
print(orders_json)
Running this script will parse the XML, convert it into a dictionary, and then transform that dictionary into JSON format. The JSON output will look like this:
{
    "Orders": {
        "Order": {
            "OrderID": "141",
            "Customer": {
                "CustomerID": "141",
                "Name": "Customer 141"
            },
            "OrderDate": "2024-10-15T10:21:00",
            "Products": null,
            "TotalAmount": "400.37"
        }
    }
}
